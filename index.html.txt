<script>
let upi="9719211706-2@ybl";
let adminPassword="admin123";
let cardsData=[];

function openShop(){
login.classList.add("hidden");
shop.classList.remove("hidden");
render();
}

function pay(price){
window.location.href=`upi://pay?pa=${upi}&am=${price}&cu=INR`;
}

function whatsapp(){
window.open("https://wa.me/91585095337?text=Hello%20AMG%20STORE,%0AMaine%20payment%20kar%20diya%20hai.%0AScreenshot%20attached");
}

function render(){
let html="";
cardsData.forEach((c,i)=>{
html+=`
<div class="card">
${c.sold?'<div class="sold">SOLD</div>':""}
<img src="${c.img}">
<div class="price">₹${c.price}</div>
<button class="buy" onclick="pay('${c.price}')">BUY NOW</button>
<img class="qr" src="${c.qr}">
<button class="buy" onclick="whatsapp()">Send Screenshot</button>
${admin.classList.contains("hidden")?"":`<button class="buy adminBtn" onclick="markSold(${i})">MARK SOLD</button>`}
</div>`;
});
cards.innerHTML=html;
}

function openAdminLogin(){
shop.classList.add("hidden");
adminLogin.classList.remove("hidden");
}

function checkPass(){
if(adminPass.value===adminPassword){
adminLogin.classList.add("hidden");
admin.classList.remove("hidden");
render();
}else alert("Wrong Password");
}

function addCard(){
cardsData.push({img:imgUrl.value,price:price.value,qr:qrUrl.value,sold:false});
imgUrl.value=price.value=qrUrl.value="";
render();
}

function markSold(i){
cardsData[i].sold=true;
setTimeout(()=>{cardsData.splice(i,1);render()},1500);
}

function changePassword(){
adminPassword=newPass.value;
newPass.value="";
alert("Password Changed");
}

function backShop(){
admin.classList.add("hidden");
shop.classList.remove("hidden");
render();
}
